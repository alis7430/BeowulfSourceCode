tree("Root")
	parallel
		repeat mute tree("Die")
		repeat mute tree("BeAlive")
		

tree("BeAlive")
	sequence
			not IsDead
		fallback
			tree("Attacked")
			//tree("ReturnToPlace")
			tree("Attack")
			tree("LookTarget")
			tree("Patroll")

tree("Attacked")
	while IsAttacked
		repeat
			sequence
				AttackedImpact

tree("ReturnToPlace")
	while IsOutOfBounds
		repeat
			sequence
				SetDestination
				parallel
					MoveToDestination

tree("LookTarget")
	while
		sequence
			IsVisibleTarget
			not HasEnemy
			not IsAttacked
		repeat
			sequence
				SetDestination_Enemy
				parallel
					LookAtDestination
					IsTargetThreaten(3.5)

tree("Patroll")
	while
		sequence
			not HasEnemy
			not IsVisibleTarget
			not IsAttacked

		repeat
			sequence
					SetRandomDestination
					MoveToDestination(7.0)
					Idle(1.0)

tree("Attack")
	while 
		sequence
			HasEnemy
			not IsDead
		fallback
			repeat
				sequence
					IsTargetInAttackRange
					WaitIdle(0.5)
					AttackTarget(1.8)
					WaitIdle(0.75)
			repeat
				sequence
					tree("ChaseTarget")
			
tree("ChaseTarget")
	while
		sequence
			not IsTargetInAttackRange
		parallel
			repeat
				SetDestination_ChaseTarget
			fallback
				repeat
					sequence
						not IsVisibleTarget
						LookAtDestination
				repeat
					MoveToDestination


tree("Die")
	sequence
		IsDead
		DeadAndDestroy(4.0)
